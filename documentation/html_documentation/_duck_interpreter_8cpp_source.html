459 double DuckInterpreter::EvaluateArithmenticExpression(const string &a_statement, int a_nextPos)
460 {
461  // Position holder for function ParseNextElement.
462  string stringValue;
463  double numValue;
464 
465  // Temp variable to store variable value. Used by GetVariableValue.
466  double temp;
467 
468  // Loop till we reach end of the statement.
469  while(a_nextPos!=INT_MAX)
470  {
471  // Get next element.
472  a_nextPos = ParseNextElement(a_statement, a_nextPos, stringValue, numValue);
473 
474  // Checking for separate semi-colons and unary not operator.
475  if(stringValue == ";" || stringValue == "!")
476  continue;
477 
478  // Checking for a string value
479  if(numValue==INT_MAX)
480  {
481  // Checking for opening brace.
482  if(stringValue == "(")
483  m_operatorStack.push_back(stringValue);
484 
485  // Checking if it is a variable.
486  else if(m_symbolTable.GetVariableValue(stringValue, temp) == true)
487  // Pushing the value in temp to the stack.
488  m_numberStack.push_back(temp);
489 
490  // Solving brace if closing brace found.
491  else if (stringValue == ")")
492  {
493  // Performing value1 (t_operator) value2
494  while(!m_operatorStack.empty() && m_operatorStack.back() != "(")
495  {
496  // Getting values from the stacks.
497  int value2 = m_numberStack.back();
498  m_numberStack.pop_back();
499 
500  int value1 = m_numberStack.back();
501  m_numberStack.pop_back();
502 
503  string t_operator = m_operatorStack.back();
504  m_operatorStack.pop_back();
505 
506  m_numberStack.push_back(DoOperation(value1, value2, t_operator));
507  }
508 
509  // Remove opening brace
510  m_operatorStack.pop_back();
511  }
512  // An operator.
513  else
514  {
515  // While the top operator has same or greater precedence than the current operator.
516  // We perform t_operator to the top two values on stack.
517  while(!m_operatorStack.empty() && GetOperatorPrecedence(m_operatorStack.back())
518  >= GetOperatorPrecedence(stringValue))
519  {
520  int value2 = m_numberStack.back();
521  m_numberStack.pop_back();
522 
523  int value1 = m_numberStack.back();
524  m_numberStack.pop_back();
525 
526  string t_operator = m_operatorStack.back();
527  m_operatorStack.pop_back();
528 
529  m_numberStack.push_back(DoOperation(value1, value2, t_operator));
530  }
531  // Push current element to stack, which is the result of the operation performed.
532  m_operatorStack.push_back(stringValue);
533  }
534  }
535  // Else it is a numeric value
536  else
537  m_numberStack.push_back(numValue);
538  }
539 
540  // Checking for any remaining operations in the operator stack.
541  if(!m_operatorStack.empty())
542  {
543  // Performing the top operation of m_operatorStack on the top two values of m_numberStack.
544  while (m_operatorStack.back() != "\0" && !m_operatorStack.empty())
545  {
546  int value2 = m_numberStack.back();
547  m_numberStack.pop_back();
548 
549  int value1 = m_numberStack.back();
550  m_numberStack.pop_back();
551 
552  string t_operator = m_operatorStack.back();
553  m_operatorStack.pop_back();
554 
555  m_numberStack.push_back(DoOperation(value1, value2, t_operator));
556  }
557  }
558 
559  // The top contains the final result.
560  return m_numberStack.back();
561 }
